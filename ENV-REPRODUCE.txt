2. Quick setup (online, recommended order)
================================================================================

# 1) Create environment
conda create -n FoundDiff python=3.10.20 -y
conda activate FoundDiff

# 2) PyTorch GPU (RTX 50 series needs cu128; RTX 30/40 can use cu121/cu124)
pip uninstall -y torch torchvision torchaudio
pip install torch==2.11.0 torchvision==0.26.0 torchaudio==2.11.0 \
  --index-url https://download.pytorch.org/whl/cu128

# 3) Pin numpy/scipy (avoid pywt/scipy ufunc errors)
pip install "numpy>=2.0,<2.3" scipy==1.15.3

# 4) Common FoundDiff pip dependencies
pip install -r requirements-founddiff-pip.txt

# 5) CLIP
pip install git+https://github.com/openai/CLIP.git

# 6) nvcc (for mamba compile; do not conda install pytorch over pip torch)
conda install -y -c nvidia cuda-nvcc=12.8

# 7) Compile mamba (RTX 5060 example; use 8.0 for A100, etc.)
export CUDA_HOME="$CONDA_PREFIX"
export PATH="$CONDA_PREFIX/bin:$PATH"
export TORCH_CUDA_ARCH_LIST="12.0"
pip install ninja packaging
pip install causal-conv1d>=1.2.0 --no-build-isolation
pip install mamba-ssm --no-build-isolation

# 8) Verify torch was not replaced
python -c "import torch; print(torch.__version__, torch.cuda.is_available())"
python -c "import selective_scan_cuda; print('mamba OK')"

# 9) If mamba install upgraded torch, reinstall torch and retest mamba
pip install torch==2.11.0 torchvision==0.26.0 torchaudio==2.11.0 \
  --index-url https://download.pytorch.org/whl/cu128

3. Offline setup
================================================================================

Online machine:
  pip download torch==2.11.0 torchvision==0.26.0 torchaudio==2.11.0 \
    --index-url https://download.pytorch.org/whl/cu128 -d wheels/
  pip download -r requirements-founddiff-pip.txt -d wheels/
  # mamba/causal-conv1d must be compiled on target with nvcc, or copy built .so wheels

Offline machine:
  conda create -n FoundDiff python=3.10.20 -y
  conda activate FoundDiff
  pip install --no-index --find-links=wheels/ torch torchvision torchaudio
  pip install --no-index --find-links=wheels/ -r requirements-founddiff-pip.txt
  # then compile or offline-install mamba wheels
